## Infos Module

Get general information about the service (e.g., country, currency).

```ts
const info = await deezer.infos();
```

::tabs
  :::tab{label="Schema"}
    <!-- sync-interface:Infos -->

    ```ts
    export interface Infos {
      country_iso: string;
      country: string;
      open: boolean;
      pop: string;
      upload_token: string;
      upload_token_lifetime: number;
      user_token: string | null;
      hosts: {
        stream: string;
        images: string;
      };
      ads: {
        audio?: {
          default: {
            start: number;
            interval: number;
            unit: string;
          };
        };
        display?: {
          interstitial: {
            start: number;
            interval: number;
            unit: string;
          };
        };
        [key: string]: any;
      };
      has_podcasts: boolean;
      offers: any[];
    }
    ```

    <!-- /sync-interface -->
  :::

## Options Module

Get available options for the API.

```ts
const options = await deezer.options();
```

::tabs
  :::tab{label="Schema"}
    <!-- sync-interface:Options -->

    ```ts
    export interface Options {
      streaming: boolean;
      streaming_duration: number;
      offline: boolean;
      hq: boolean;
      ads_display: boolean;
      ads_audio: boolean;
      too_many_devices: boolean;
      can_subscribe: boolean;
      radio_skips: number;
      lossless: boolean;
      preview: boolean;
      radio: boolean;
      type: "options";
    }
    ```

    <!-- /sync-interface -->
  :::
